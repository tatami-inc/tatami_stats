82void count_running(const bool row, const tatami::Matrix<Value_, Index_>& mat, Output_* const output, Condition_ condition, const CountOptions& copt) {
83 const Index_ dim = (row ? mat.nrow() : mat.ncol());
84 const Index_ otherdim = (row ? mat.ncol() : mat.nrow());
85
86 const bool do_parallel = copt.num_threads > 1;
87 std::optional<std::vector<std::optional<std::vector<Output_> > > > all_partial_count;
88 if (do_parallel) {
89 all_partial_count.emplace(sanisizer::cast<I<decltype(all_partial_count->size())> >(copt.num_threads - 1));
90 }
91
92 // Checking if we should count zeros in the sparse case.
93 const bool is_sparse = mat.is_sparse();
94 const bool count_zero = is_sparse && condition(0);
95
96 std::fill_n(output, dim, 0);
97
98 const int num_used = tatami::parallelize([&](int thread, Index_ start, Index_ len) -> void {
99 Output_* out_ptr;
100 std::optional<std::vector<Output_> > cur_count;
101 if (!do_parallel) {
102 out_ptr = output;
103 } else {
104 // Directly write the result to the output buffer for the first thread to save ourselves an allocation.
105 // Everything else goes into these temporary vectors.
106 if (thread == 0) {
107 out_ptr = output;
108 } else {
109 cur_count.emplace(tatami::cast_Index_to_container_size<std::vector<Output_> >(dim));
110 out_ptr = cur_count->data();
111 }
112 }
113
114 if (is_sparse) {
115 tatami::Options opt;
116 opt.sparse_ordered_index = false;
117
118 auto xbuffer = tatami::create_container_of_Index_size<std::vector<Value_> >(dim);
119 auto ibuffer = tatami::create_container_of_Index_size<std::vector<Index_> >(dim);
120 auto ext = tatami::consecutive_extractor<true>(mat, !row, start, len, opt);
121 auto nonzeros = tatami::create_container_of_Index_size<std::vector<Index_> >(dim);
122
123 for (Index_ x = 0; x < len; ++x) {
124 auto range = ext->fetch(xbuffer.data(), ibuffer.data());
125 for (Index_ j = 0; j < range.number; ++j) {
126 auto idx = range.index[j];
127 out_ptr[idx] += condition(range.value[j]);
128 ++(nonzeros[idx]);
129 }
130 }
131
132 if (count_zero) {
133 for (Index_ d = 0; d < dim; ++d) {
134 out_ptr[d] += len - nonzeros[d];
135 }
136 }
137
138 } else {
139 auto xbuffer = tatami::create_container_of_Index_size<std::vector<Value_> >(dim);
140 auto ext = tatami::consecutive_extractor<false>(mat, !row, start, len);
141
142 for (Index_ x = 0; x < len; ++x) {
143 auto ptr = ext->fetch(xbuffer.data());
144 for (Index_ d = 0; d < dim; ++d) {
145 out_ptr[d] += condition(ptr[d]);
146 }
147 }
148 }
149
150 if (thread) {
151 (*all_partial_count)[thread - 1] = std::move(cur_count);
152 }
153 }, otherdim, copt.num_threads);
154
155 if (do_parallel) {
156 // Skip the first thread as we already put its counts in 'output'.
157 for (int u = 1; u < num_used; ++u) {
158 const auto& curout = *((*all_partial_count)[u - 1]);
159 for (Index_ d = 0; d < dim; ++d) {
160 output[d] += curout[d];
161 }
162 }
163 }
