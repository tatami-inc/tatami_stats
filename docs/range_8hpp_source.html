40template<typename Value_>
41constexpr Value_ choose_minimum_placeholder() {
42 // Placeholder value 'x' is such that 'x >= y' is always true for any non-NaN 'y'.
43 if constexpr(std::numeric_limits<Value_>::has_infinity) {
44 return std::numeric_limits<Value_>::infinity();
45 } else {
46 return std::numeric_limits<Value_>::max();
47 }
48}

50template<typename Value_>
51constexpr Value_ choose_maximum_placeholder() {
52 // Placeholder value 'x' is such that 'x <= y' is always true for any non-NaN 'y'.
53 if constexpr(std::numeric_limits<Value_>::has_infinity) {
54 return -std::numeric_limits<Value_>::infinity();
55 } else {
56 return std::numeric_limits<Value_>::lowest();
57 }

60template<typename Value_, typename Index_>
61Value_ min_direct(const Value_* const ptr, const Index_ num, bool skip_nan) {
62 return internal::nanable_ifelse_with_value<Value_>(
63 skip_nan,
64 [&]() -> Value_ {
65 auto current = choose_minimum_placeholder<Value_>();
66 for (Index_ i = 0; i < num; ++i) {
67 auto val = ptr[i];
68 if (val < current) { // no need to explicitly handle NaNs, as any comparison with NaNs is always false.
69 current = val;
70 }
71 }
72 return current;
73 },
74 [&]() -> Value_ {
75 if (num) {
76 return *std::min_element(ptr, ptr + num);
77 } else {
78 return choose_minimum_placeholder<Value_>();
79 }
80 }
81 );
82}

169template<typename Value_, typename Index_, typename Output_>
170void range_direct(bool row, const tatami::Matrix<Value_, Index_>& mat, RangeBuffers<Output_>& output, const RangeOptions& ropt) {
171 const auto dim = (row ? mat.nrow() : mat.ncol());
172 const auto otherdim = (row ? mat.ncol() : mat.nrow());
173
174 if (mat.is_sparse()) {
175 tatami::Options opt;
176 opt.sparse_extract_index = false;
177 tatami::parallelize([&](int, Index_ s, Index_ l) -> void {
178 auto ext = tatami::consecutive_extractor<true>(mat, row, s, l, opt);
179 auto vbuffer = tatami::create_container_of_Index_size<std::vector<Value_> >(otherdim);
180 for (Index_ x = 0; x < l; ++x) {
181 auto out = ext->fetch(vbuffer.data(), NULL);
182 output.minimum[x + s] = min_direct(out.value, out.number, otherdim, ropt.skip_nan);
183 output.maximum[x + s] = max_direct(out.value, out.number, otherdim, ropt.skip_nan);
184 }
185 }, dim, ropt.num_threads);
186
187 } else {
188 tatami::parallelize([&](int, Index_ s, Index_ l) -> void {
189 auto ext = tatami::consecutive_extractor<false>(mat, row, s, l);
190 auto buffer = tatami::create_container_of_Index_size<std::vector<Value_> >(otherdim);
191 for (Index_ x = 0; x < l; ++x) {
192 auto ptr = ext->fetch(buffer.data());
193 output.minimum[x + s] = min_direct(ptr, otherdim, ropt.skip_nan);
194 output.maximum[x + s] = max_direct(ptr, otherdim, ropt.skip_nan);
195 }
196 }, dim, ropt.num_threads);
197 }

201void range_running(bool row, const tatami::Matrix<Value_, Index_>& mat, RangeBuffers<Output_>& output, const RangeOptions& ropt) {
202 const auto dim = (row ? mat.nrow() : mat.ncol());
203 const auto otherdim = (row ? mat.ncol() : mat.nrow());
204 const bool is_sparse = mat.is_sparse();
205
206 const bool do_parallel = ropt.num_threads > 1;
207 std::optional<std::vector<std::optional<std::vector<Output_> > > > all_partial_min, all_partial_max;
208 if (do_parallel) {
209 all_partial_min.emplace(sanisizer::cast<I<decltype(all_partial_min->size())> >(ropt.num_threads - 1));
210 all_partial_max.emplace(sanisizer::cast<I<decltype(all_partial_max->size())> >(ropt.num_threads - 1));
211 }
212
213 constexpr auto min_placeholder = choose_minimum_placeholder<Value_>();
214 constexpr auto max_placeholder = choose_maximum_placeholder<Value_>();
215 std::fill_n(output.minimum, dim, min_placeholder);
216 std::fill_n(output.maximum, dim, max_placeholder);
217
218 const auto nused = tatami::parallelize([&](int thread, Index_ s, Index_ l) -> void {
219 Output_* min_ptr;
220 Output_* max_ptr;
221 std::optional<std::vector<Output_> > cur_min, cur_max;
222 if (!do_parallel) {
223 min_ptr = output.minimum;
224 max_ptr = output.maximum;
225 } else {
226 if (thread == 0) {
227 min_ptr = output.minimum;
228 max_ptr = output.maximum;
229 } else {
230 cur_min.emplace(tatami::cast_Index_to_container_size<std::vector<Output_> >(dim), min_placeholder);
231 cur_max.emplace(tatami::cast_Index_to_container_size<std::vector<Output_> >(dim), max_placeholder);
232 min_ptr = cur_min->data();
233 max_ptr = cur_max->data();
234 }
235 }
236
237 if (is_sparse) {
238 tatami::Options opt;
239 opt.sparse_ordered_index = false;
240 auto ext = tatami::consecutive_extractor<true>(mat, !row, s, l, opt);
241 auto vbuffer = tatami::create_container_of_Index_size<std::vector<Value_> >(dim);
242 auto ibuffer = tatami::create_container_of_Index_size<std::vector<Index_> >(dim);
243 auto nonzeros = tatami::create_container_of_Index_size<std::vector<Index_> >(dim);
244
245 for (Index_ x = 0; x < l; ++x) {
246 auto out = ext->fetch(vbuffer.data(), ibuffer.data());
247
248 if (x == 0) {
249 // For the first observed vector in each thread,
250 // we can optimize it a little as we don't need to read existing min/max.
251 internal::nanable_ifelse<Value_>(
252 ropt.skip_nan,
253 [&]() -> void {
254 for (Index_ i = 0; i < out.number; ++i) {
255 const auto val = out.value[i];
256 if (!std::isnan(val)) {
257 const auto idx = out.index[i];
258 min_ptr[idx] = val;
259 max_ptr[idx] = val;
260 ++nonzeros[idx];
261 }
262 }
263 },
264 [&]() -> void {
265 for (Index_ i = 0; i < out.number; ++i) {
266 const auto val = out.value[i];
267 const auto idx = out.index[i];
268 min_ptr[idx] = val;
269 max_ptr[idx] = val;
270 ++nonzeros[idx];
271 }
272 }
273 );
274
275 } else {
276 for (Index_ i = 0; i < out.number; ++i) {
277 const auto val = out.value[i];
278 const auto idx = out.index[i];
279 auto& min_current = min_ptr[idx];
280 if (val < min_current) { // this should implicitly skip val=NaNs, as any NaN comparison will be false.
281 min_current = val;
282 }
283 auto& max_current = max_ptr[idx];
284 if (val > max_current) { // this should implicitly skip val=NaNs, as any NaN comparison will be false.
285 max_current = val;
286 }
287 ++nonzeros[idx];
288 }
289 }
290 }

293 if (l > nonzeros[d]) {
294 auto& min_current = min_ptr[d];
295 if (min_current > 0) {
296 min_current = 0;
297 }
298 auto& max_current = max_ptr[d];
299 if (max_current < 0) {
300 max_current = 0;
301 }
302 }
303 }
304
305 } else {
306 auto ext = tatami::consecutive_extractor<false>(mat, !row, s, l);
307 auto buffer = tatami::create_container_of_Index_size<std::vector<Value_> >(dim);
308
309 for (Index_ x = 0; x < l; ++x) {
310 auto ptr = ext->fetch(buffer.data());
311
312 if (x == 0) {
313 // For the first observed vector in each thread,
314 // we can optimize it a little as we don't need to read existing min/max.
315 internal::nanable_ifelse<Value_>(
316 ropt.skip_nan,
317 [&]() -> void {
318 for (Index_ i = 0; i < dim; ++i) {
319 const auto val = ptr[i];
320 if (!std::isnan(val)) {
321 min_ptr[i] = val;
322 max_ptr[i] = val;
323 }
324 }

334 const auto val = ptr[i];

362 output.minimum[d] = cur_min[d];
363 }
364 if (output.maximum[d] < cur_max[d]) {
365 output.maximum[d] = cur_max[d];
366 }
367 }
368 }
369 }
