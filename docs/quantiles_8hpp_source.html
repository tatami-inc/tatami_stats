60template<typename Output_, typename Value_, typename Index_>
61class SingleQuantile {
62public:
68 SingleQuantile(const Index_ len, const Output_ quantile) : my_len(len) {
69 assert(len > 0);
70 static_assert(std::is_integral<Index_>::value);
71 static_assert(std::is_floating_point<Output_>::value);
72
73 if (quantile < 0 || quantile > 1) {
74 throw std::out_of_range("'quantile' should lie in [0, 1]");
75 }
76
77 const Index_ len_m1 = len - 1;
78 const Output_ raw = static_cast<Output_>(len_m1) * quantile;
79 const Output_ upper_index = std::ceil(raw);
80 const Output_ lower_index = std::floor(raw);
81
82 // Protect the cast from conversion imprecision between size_t and Float_,
83 // e.g., if it rounds up or if it converts it to an Inf.
84 const auto upper_san = sanisizer::from_float<Index_>(upper_index);
85 if (upper_san <= len_m1) {
86 my_upper_index = upper_san;
87 my_upper_fraction = raw - lower_index;
88 my_skip_lower = (upper_index == lower_index);
89 } else {
90 // Just gently cap it at the maximum value.
91 my_upper_index = len_m1;
92 my_upper_fraction = 0;
93 my_skip_lower = true;
94 }
95 }
96
97private:
98 Index_ my_len, my_upper_index;
99 Output_ my_upper_fraction;
100 bool my_skip_lower;
101
102public:
112 Output_ operator()(Value_* ptr) const {
113 const auto target = ptr + my_upper_index;
114 std::nth_element(ptr, target, ptr + my_len);
115
116 // Avoid an extra memory access to get the lower index if we don't need it; this would also fail if upper_index = 0.
117 const Output_ upper = *target;
118 if (my_skip_lower) {
119 return upper;
120 }
121
122 const Output_ lower = *std::max_element(ptr, target);
123 return lower + (upper - lower) * my_upper_fraction;
124 }
125
137 Output_ operator()(const Index_ num, Value_* ptr) const {
138 assert(num <= my_len);
139 if (num == my_len) {
140 return operator()(ptr);
141 } else if (num == 0) {
142 return 0;
143 }
144
145 Index_ num_negative = 0;
146 for (Index_ i = 0; i < num; ++i) {
147 num_negative += (ptr[i] < 0);
148 }
149
150 if (my_upper_index < num_negative) {
151 const auto target = ptr + my_upper_index;
152 std::nth_element(ptr, target, ptr + num);
153
154 const Output_ upper = *target;
155 if (my_skip_lower) {
156 return upper;
157 }
158
159 const Output_ lower = *std::max_element(ptr, target);
160 return lower + (upper - lower) * my_upper_fraction;
161 }
162
163 if (num_negative && my_upper_index == num_negative) {
164 // The upper value is zero. It can't be positive, as that would
165 // imply that there are no structural zeros and num == my_len.
166 if (my_skip_lower) {
167 return 0;
168 }
169
170 const auto target = ptr + (my_upper_index - 1);
171 std::nth_element(ptr, target, ptr + num);
172 return static_cast<Output_>(*target) * (1 - my_upper_fraction);
173 }
174
175 const Index_ num_zeros = my_len - num;
176 const Index_ num_not_positive = num_zeros + num_negative;
177 if (my_upper_index < num_not_positive) {
178 return 0;
179 }
180
181 const auto target = ptr + (my_upper_index - num_zeros);
182 std::nth_element(ptr, target, ptr + num);
183 const Output_ upper = *target;
184 if (my_skip_lower) {
185 return upper;
186 }
187
188 if (sanisizer::is_equal(my_upper_index, num_not_positive)) {
189 return upper * my_upper_fraction;
190 }
191
192 const Output_ lower = *std::max_element(ptr, target);
193 return lower + (upper - lower) * my_upper_fraction;
194 }

219 SingleQuantileVariableLength(Index_ max_len, const double quantile) : my_quantile(quantile) {

247 Output_ operator()(const Index_ len, Value_* ptr) {

249 return std::numeric_limits<Output_>::quiet_NaN();

315void apply(bool row, const tatami::Matrix<Value_, Index_>& mat, const double quantile, Output_* output, const Options& qopt) {
316 auto dim = (row ? mat.nrow() : mat.ncol());
317 auto otherdim = (row ? mat.ncol() : mat.nrow());
318
319 if (otherdim == 0) {
320 // Prevent initalize() from constructing with a fixed instance with otherdim == 0.
321 std::fill_n(output, dim, std::numeric_limits<Output_>::quiet_NaN());
322 return;
323 }
324
325 auto initialize = [&](
326 std::optional<SingleQuantile<Output_, Value_, Index_> >& fixed,
327 std::optional<SingleQuantileVariableLength<Output_, Value_, Index_> >& variable
328 ) -> void {
329 if (qopt.skip_nan) {
330 variable.emplace(otherdim, quantile);
331 } else {
332 fixed.emplace(otherdim, quantile);
333 }
334 };
335
336 if (mat.sparse()) {
337 tatami::Options opt;
338 opt.sparse_extract_index = false;
339 opt.sparse_ordered_index = false; // we'll be sorting by value anyway.
340
341 tatami::parallelize([&](int, Index_ s, Index_ l) -> void {
342 auto ext = tatami::consecutive_extractor<true>(mat, row, s, l, opt);
343 auto buffer = tatami::create_container_of_Index_size<std::vector<Value_> >(otherdim);
344 auto vbuffer = buffer.data();
345
346 std::optional<SingleQuantile<Output_, Value_, Index_> > qcalcs_fixed;
347 std::optional<SingleQuantileVariableLength<Output_, Value_, Index_> > qcalcs_var;
348 initialize(qcalcs_fixed, qcalcs_var);
349
350 for (Index_ x = 0; x < l; ++x) {
351 auto range = ext->fetch(vbuffer, NULL);
352 tatami::copy_n(range.value, range.number, vbuffer);
353
354 ::tatami_stats::internal::nanable_ifelse<Value_>(
355 qopt.skip_nan,
356 [&]() -> void {
357 auto lost = shift_nans(vbuffer, range.number);
358 output[x + s] = (*qcalcs_var)(otherdim - lost, range.number - lost, vbuffer + lost);
359 },
360 [&]() -> void {
361 output[x + s] = (*qcalcs_fixed)(range.number, vbuffer);
362 }
363 );
364
365 }
366 }, dim, qopt.num_threads);
367
368 } else {
369 tatami::parallelize([&](int, Index_ s, Index_ l) -> void {
370 auto buffer = tatami::create_container_of_Index_size<std::vector<Value_> >(otherdim);
371 auto ext = tatami::consecutive_extractor<false>(mat, row, s, l);
372
373 std::optional<SingleQuantile<Output_, Value_, Index_> > qcalcs_fixed;
374 std::optional<SingleQuantileVariableLength<Output_, Value_, Index_> > qcalcs_var;
375 initialize(qcalcs_fixed, qcalcs_var);
376
377 for (Index_ x = 0; x < l; ++x) {
378 auto bufptr = buffer.data();
379 auto raw = ext->fetch(bufptr);
380 tatami::copy_n(raw, otherdim, bufptr);
381
382 ::tatami_stats::internal::nanable_ifelse<Value_>(
383 qopt.skip_nan,
384 [&]() -> void {
385 auto lost = shift_nans(bufptr, otherdim);
386 output[x + s] = (*qcalcs_var)(otherdim - lost, bufptr + lost);
387 },
388 [&]() -> void {
389 output[x + s] = (*qcalcs_fixed)(bufptr);
390 }
391 );
392 }
393 }, dim, qopt.num_threads);
394 }
