36

39
40 if (p->sparse()) {

42 opt.sparse_ordered_index = false;
43

45 std::vector<Index_> ibuffer(otherdim);
46 for (Index_ i = 0; i < len; ++i) {
47 auto range = ext->fetch(xbuffer.data(), ibuffer.data());
48 for (Index_ j = 0; j < range.number; ++j) {
49 workspace[groups[range.index[j]]].push_back(range.value[j]);
50 }
51
52 for (size_t g = 0; g < ngroups; ++g, ++curoutput) {

55 w.clear();
56 }
57 }
58
59 } else {

61 for (Index_ i = 0; i < len; ++i) {

64 workspace[groups[j]].push_back(ptr[j]);

66
