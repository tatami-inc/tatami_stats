46template<bool minimum_, typename Value_>
47constexpr auto choose_placeholder() {
48 if constexpr(minimum_) {
49 // Placeholder value 'x' is such that 'x > y' is always true for any non-NaN 'y'.
50 if constexpr(std::numeric_limits<Value_>::has_infinity) {
51 return std::numeric_limits<Value_>::infinity();
52 } else {
53 return std::numeric_limits<Value_>::max();
54 }
55 } else {
56 // Placeholder value 'x' is such that 'x < y' is always true for any non-NaN 'y'.
57 if constexpr(std::numeric_limits<Value_>::has_infinity) {
58 return -std::numeric_limits<Value_>::infinity();
59 } else {
60 return std::numeric_limits<Value_>::lowest();
61 }
62 }
63}

65template<bool minimum_, typename Output_, typename Value_>
66bool is_better(Output_ best, Value_ alt) {
67 if constexpr(minimum_) {
68 return best > static_cast<Output_>(alt);
69 } else {
70 return best < static_cast<Output_>(alt);
71 }

95template<bool minimum_, typename Value_, typename Index_>
96Value_ direct(const Value_* ptr, Index_ num, bool skip_nan) {
97 if (skip_nan) {
98 auto current = internal::choose_placeholder<minimum_, Value_>();
99 for (Index_ i = 0; i < num; ++i) {
100 auto val = ptr[i];
101 if (internal::is_better<minimum_>(current, val)) { // no need to explicitly handle NaNs, as any comparison with NaNs is always false.
102 current = val;
103 }
104 }
105 return current;
106
107 } else if (num) {
108 if constexpr(minimum_) {
109 return *std::min_element(ptr, ptr + num);
110 } else {
111 return *std::max_element(ptr, ptr + num);
112 }
113
114 } else {
115 return internal::choose_placeholder<minimum_, Value_>();
116 }
117}

137template<bool minimum_, typename Value_, typename Index_>
138Value_ direct(const Value_* value, Index_ num_nonzero, Index_ num_all, bool skip_nan) {
139 if (num_nonzero) {
140 auto candidate = direct<minimum_>(value, num_nonzero, skip_nan);
141 if (num_nonzero < num_all && internal::is_better<minimum_>(candidate, 0)) {
142 candidate = 0;
143 }
144 return candidate;
145 } else if (num_all) {
146 return 0;
147 } else {
148 return internal::choose_placeholder<minimum_, Value_>();

168class RunningDense {

176 RunningDense(Index_ num, Output_* store, bool skip_nan) : my_num(num), my_store(store), my_skip_nan(skip_nan) {}

182 void add(const Value_* ptr) {
183 if (my_init) {
184 my_init = false;
185 if (my_skip_nan) {
186 for (Index_ i = 0; i < my_num; ++i, ++ptr) {
187 auto val = *ptr;

192 }
193 }

197
198 } else {
199 for (Index_ i = 0; i < my_num; ++i, ++ptr) {
200 auto val = *ptr;
201 if (internal::is_better<minimum_>(my_store[i], val)) { // this should implicitly skip NaNs, any NaN comparison will be false.
202 my_store[i] = val;
203 }
204 }
205 }
206 }

211 void finish() {
212 if (my_init) {
213 std::fill_n(my_store, my_num, internal::choose_placeholder<minimum_, Value_>());
214 }
215 }

217private:

236template<bool minimum_, typename Output_, typename Value_, typename Index_>
237class RunningSparse {
238public:
248 RunningSparse(Index_ num, Output_* store, bool skip_nan, Index_ subtract = 0) :
249 my_num(num), my_store(store), my_skip_nan(skip_nan), my_subtract(subtract) {}
250
257 void add(const Value_* value, const Index_* index, Index_ number) {
258 if (my_count == 0) {
259 my_nonzero.resize(my_num);
260 std::fill_n(my_store, my_num, internal::choose_placeholder<minimum_, Value_>());
261
262 if (!my_skip_nan) {
263 for (Index_ i = 0; i < number; ++i, ++value, ++index) {
264 auto val = *value;
265 auto idx = *index - my_subtract;
266 my_store[idx] = val;
267 ++my_nonzero[idx];
268 }
269 my_count = 1;
270 return;
271 }
272 }
273
274 for (Index_ i = 0; i < number; ++i, ++value, ++index) {
275 auto val = *value;
276 auto idx = *index - my_subtract;
277 auto& current = my_store[idx];
278 if (internal::is_better<minimum_>(current, val)) { // this should implicitly skip NaNs, any NaN comparison will be false.
279 current = val;
280 }
281 ++my_nonzero[idx];
282 }
283
284 ++my_count;
285 }
286
290 void finish() {
291 if (my_count) {
292 for (Index_ i = 0; i < my_num; ++i) {
293 if (my_count > my_nonzero[i]) {
294 auto& current = my_store[i];
295 if (internal::is_better<minimum_>(current, 0)) {
296 current = 0;
297 }
298 }
299 }
300 } else {
301 std::fill_n(my_store, my_num, internal::choose_placeholder<minimum_, Value_>());
302 }
303 }
304
305private:
306 Index_ my_num;
307 Output_* my_store;
308 bool my_skip_nan;
309 Index_ my_subtract;
310 Index_ my_count = 0;
311 std::vector<Index_> my_nonzero;
