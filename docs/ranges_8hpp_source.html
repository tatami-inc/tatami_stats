22template<bool row_, typename Output_, typename Value_, typename Index_, class StoreMinimum_, class StoreMaximum_>
23void dimension_extremes(const tatami::Matrix<Value_, Index_>* p, int threads, StoreMinimum_& min_out, StoreMaximum_& max_out) {
24 auto dim = (row_ ? p->nrow() : p->ncol());
25 auto otherdim = (row_ ? p->ncol() : p->nrow());
26 const bool direct = p->prefer_rows() == row_;
27
28 constexpr bool store_min = !std::is_same<StoreMinimum_, bool>::value;
29 constexpr bool store_max = !std::is_same<StoreMaximum_, bool>::value;
30
31 if (!otherdim) {
32 if constexpr(store_min) {
33 std::fill(min_out, min_out + dim, 0);
34 }
35 if constexpr(store_max) {
36 std::fill(max_out, max_out + dim, 0);
37 }
38 return;
39 }
40
41 if (p->sparse()) {
42 tatami::Options opt;
43 opt.sparse_ordered_index = false;
44
45 if (direct) {
46 opt.sparse_extract_index = false;
47 tatami::parallelize([&](size_t, Index_ s, Index_ l) {
48 auto ext = tatami::consecutive_extractor<true>(p, row_, s, l, opt);
49 std::vector<Value_> vbuffer(otherdim);
50
51 for (Index_ x = 0; x < l; ++x) {
52 auto out = ext->fetch(vbuffer.data(), NULL);
53 if (out.number) {
54 if constexpr(store_min) {
55 auto minned = *std::min_element(out.value, out.value + out.number);
56 if (minned > 0 && out.number != otherdim) {
57 minned = 0;
58 }
59 min_out[x + s] = minned;
60 }
61 if constexpr(store_max) {
62 auto maxed = *std::max_element(out.value, out.value + out.number);
63 if (maxed < 0 && out.number != otherdim) {
64 maxed = 0;
65 }
66 max_out[x + s] = maxed;
67 }
68 } else {
69 if constexpr(store_min) {
70 min_out[x + s] = 0;
71 }
72 if constexpr(store_max) {
73 max_out[x + s] = 0;
74 }
75 }
76 }
77 }, dim, threads);
78
79 } else {
80 tatami::parallelize([&](size_t, Index_ s, Index_ l) {
81 auto ext = tatami::consecutive_extractor<true>(p, !row_, 0, otherdim, s, l, opt);
82 std::vector<Value_> vbuffer(l);
83 std::vector<Index_> ibuffer(l);
84 std::vector<Index_> counter(l);
85
86 for (Index_ x = 0; x < otherdim; ++x) {
87 auto out = ext->fetch(vbuffer.data(), ibuffer.data());
88 for (Index_ j = 0; j < out.number; ++j) {
89 auto idx = out.index[j];
90 auto& c = counter[idx - s];
91 auto val = static_cast<Output_>(out.value[j]);
92 if constexpr(store_min) {
93 auto& last = min_out[idx];
94 if (c == 0 || last > val) {
95 last = val;
96 }
97 }
98 if constexpr(store_max) {
99 auto& last = max_out[idx];
100 if (c == 0 || last < val) {
101 last = val;
102 }
103 }
104 ++c;
105 }
106 }
107
108 // Handling the zeros.
109 for (Index_ i = s, e = s + l; i < e; ++i) {
110 auto c = counter[i - s];
111 if (c == 0) {
112 if constexpr(store_min) {
113 min_out[i] = 0;
114 }
115 if constexpr(store_max) {
116 max_out[i] = 0;
117 }
118 } else if (c < otherdim) {
119 if constexpr(store_min) {
120 if (min_out[i] > 0) {
121 min_out[i] = 0;
122 }
123 }
124 if constexpr(store_max) {
125 if (max_out[i] < 0) {
126 max_out[i] = 0;
127 }
128 }
129 }
130 }
131 }, dim, threads);
132 }
133
134 } else {
135 if (direct) {
136 tatami::parallelize([&](size_t, Index_ s, Index_ l) {
137 auto ext = tatami::consecutive_extractor<false>(p, row_, s, l);
138 std::vector<Value_> buffer(otherdim);
139 for (Index_ x = 0; x < l; ++x) {
140 auto ptr = ext->fetch(buffer.data());
141 if constexpr(store_min) {
142 min_out[x + s] = *std::min_element(ptr, ptr + otherdim);
143 }
144 if constexpr(store_max) {
145 max_out[x + s] = *std::max_element(ptr, ptr + otherdim);
146 }
147 }
148 }, dim, threads);
149
150 } else {
151 tatami::parallelize([&](size_t, Index_ s, Index_ l) {
152 auto ext = tatami::consecutive_extractor<false>(p, !row_, 0, otherdim, s, l);
153 std::vector<Value_> buffer(l);
154
155 // We already have a otherdim > 0 check above.
156 {
157 auto ptr = ext->fetch(buffer.data());
158 if constexpr(store_min) {
159 std::copy(ptr, ptr + l, min_out + s);
160 }
161 if constexpr(store_max) {
162 std::copy(ptr, ptr + l, max_out + s);
163 }
164 }
165
166 for (Index_ x = 1; x < otherdim; ++x) {
167 auto ptr = ext->fetch(buffer.data());
168 for (Index_ d = 0; d < l; ++d) {
169 auto idx = d + s;
170 auto val = static_cast<Output_>(ptr[d]);
171 if constexpr(store_min) {
172 auto& last = min_out[idx];
173 if (last > val) {
174 last = val;
175 }
176 }
177 if constexpr(store_max) {
178 auto& last = max_out[idx];
179 if (last < val) {
180 last = val;
181 }
182 }
183 }
184 }
185 }, dim, threads);
186 }
187 }
188
189 return;
190}
