25template<bool get_minimum_, typename Value_>
26constexpr auto choose_placeholder() {
27 if constexpr(get_minimum_) {
28 // Placeholder value 'x' is such that 'x > y' is always true for any non-NaN 'y'.
29 if constexpr(std::numeric_limits<Value_>::has_infinity) {
30 return std::numeric_limits<Value_>::infinity();
31 } else {
32 return std::numeric_limits<Value_>::max();
33 }
34 } else {
35 // Placeholder value 'x' is such that 'x < y' is always true for any non-NaN 'y'.
36 if constexpr(std::numeric_limits<Value_>::has_infinity) {
37 return -std::numeric_limits<Value_>::infinity();
38 } else {
39 return std::numeric_limits<Value_>::lowest();
40 }
41 }
42}

45bool is_better(Output_ best, Value_ alt) {
46 if constexpr(get_minimum_) {

75template<bool get_minimum_, bool skip_nan_ = false, typename Value_, typename Index_>
76Value_ compute(const Value_* ptr, Index_ num) {
77 if constexpr(skip_nan_) {
78 auto current = internal::choose_placeholder<get_minimum_, Value_>();
79 for (Index_ i = 0; i < num; ++i) {
80 auto val = ptr[i];
81 if (internal::is_better<get_minimum_>(current, val)) { // no need to explicitly handle NaNs, as any comparison with NaNs is always false.
82 current = val;
83 }
84 }
85 return current;
86
87 } else if (num) {
88 if constexpr(get_minimum_) {
89 return *std::min_element(ptr, ptr + num);
90 } else {
91 return *std::max_element(ptr, ptr + num);
92 }
93
94 } else {
95 return internal::choose_placeholder<get_minimum_, Value_>();
96 }

118template<bool get_minimum_, bool skip_nan_ = false, typename Value_, typename Index_>
119Value_ compute(const Value_* value, Index_ num_nonzero, Index_ num_all) {
120 if (num_nonzero) {
121 auto candidate = compute<get_minimum_, skip_nan_>(value, num_nonzero);
122 if (num_nonzero < num_all && internal::is_better<get_minimum_>(candidate, static_cast<Value_>(0))) {

150template<bool get_minimum_, bool skip_nan_ = false, typename Value_ = double, typename Index_ = int, typename Output_ = double>
151struct RunningDense {
157 RunningDense(Index_ num, Output_* store) : num(num), store(store) {}
158
163 void add(const Value_* ptr) {
164 if (init) {
165 init = false;
166 for (Index_ i = 0; i < num; ++i, ++ptr) {
167 auto val = *ptr;
168 if constexpr(skip_nan_) {
169 if (std::isnan(val)) {
170 val = internal::choose_placeholder<get_minimum_, Value_>();
171 }
172 }
173 store[i] = val;
174 }
175
176 } else {
177 for (Index_ i = 0; i < num; ++i, ++ptr) {
178 auto val = *ptr;
179 if (internal::is_better<get_minimum_>(store[i], val)) { // this should implicitly skip NaNs, any NaN comparison will be false.
180 store[i] = val;
181 }
182 }
183 }
184 }
185
189 void finish() {
190 if (init) {
191 std::fill_n(store, num, internal::choose_placeholder<get_minimum_, Value_>());
192 }
193 }
194
195private:
196 bool init = true;
197 Index_ num;
198 Output_* store;

214template<bool get_minimum_, bool skip_nan_ = false, typename Value_ = double, typename Index_ = int, typename Output_ = double>
215struct RunningSparse {
224 RunningSparse(Index_ num, Output_* store, Index_ subtract = 0) : num(num), store(store), subtract(subtract) {}
225
232 void add(const Value_* value, const Index_* index, Index_ number) {
233 if (count == 0) {
234 nonzero.resize(num);
235 std::fill_n(store, num, internal::choose_placeholder<get_minimum_, Value_>());
236 }
237
238 for (Index_ i = 0; i < number; ++i, ++value, ++index) {
239 auto val = *value;
240 auto idx = *index - subtract;
241 auto& current = store[idx];
242 if (internal::is_better<get_minimum_>(current, val)) { // this should implicitly skip NaNs, any NaN comparison will be false.
243 current = val;
244 }
245 ++nonzero[idx];
246 }
247
248 ++count;
249 }
250
254 void finish() {
255 if (count) {
256 for (Index_ i = 0; i < num; ++i) {
257 if (count > nonzero[i]) {
258 auto& current = store[i];
259 if (internal::is_better<get_minimum_>(current, static_cast<Output_>(0))) {
260 current = 0;
261 }
262 }
263 }
264 } else {
265 std::fill_n(store, num, internal::choose_placeholder<get_minimum_, Value_>());
266 }
267 }

270 Index_ num;
271 Output_* store;
272 Index_ subtract;
273 Index_ count = 0;
274 std::vector<Index_> nonzero;
