38template<typename Output_, typename Value_>
39void add_neumaier(Output_& sum, Output_& error, Value_ val) {
40 auto t = sum + val;
41 if (std::abs(sum) >= std::abs(val)) {
42 error += (sum - t) + val;
43 } else {
44 error += (val - t) + sum;
45 }
46 sum = t;
47}
