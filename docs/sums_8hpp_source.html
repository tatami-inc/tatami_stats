39template<bool skip_nan_ = false, typename Output_, typename Value_>
40void add(Output_& sum, Output_& error, Value_ val) {
41 if constexpr(skip_nan_) {
42 if (std::isnan(val)) {
43 return;
44 }
45 }
46
47 auto t = sum + val;
48 if (std::abs(sum) >= std::abs(val)) {
49 error += (sum - t) + val;
50 } else {
51 error += (val - t) + sum;
52 }
53 sum = t;
54}
