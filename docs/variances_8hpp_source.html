63template<typename Output_ = double, typename Index_ >
64void add_welford_zeros(Output_& mean, Output_& sumsq, Index_ num_nonzero, Index_ num_all) {
65 auto ratio = static_cast<Output_>(num_nonzero) / static_cast<Output_>(num_all);
