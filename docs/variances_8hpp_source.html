48void add_welford(Output_& mean, Output_& sumsq, Value_ value, Index_ count) {
49 Output_ delta = value - mean;
50 mean += delta / count;

55void add_welford_zeros(Output_& mean, Output_& sumsq, Index_ num_nonzero, Index_ num_all) {
56 auto ratio = static_cast<Output_>(num_nonzero) / static_cast<Output_>(num_all);

84template<typename Output_ = double, typename Value_, typename Index_ >
85std::pair<Output_, Output_> direct(const Value_* value, Index_ num_nonzero, Index_ num_all, bool skip_nan) {
86 Output_ mean = 0;
87 Index_ lost = 0;
88
89 if (skip_nan) {
90 auto copy = value;
91 for (Index_ i = 0; i < num_nonzero; ++i, ++copy) {
92 auto val = *copy;
93 if (std::isnan(val)) {
94 ++lost;
95 } else {
96 mean += val;
97 }
98 }
99 } else {
100 auto copy = value;
101 for (Index_ i = 0; i < num_nonzero; ++i, ++copy) {
102 mean += *copy;
103 }
104 }
105
106 auto count = num_all - lost;
107 mean /= count;
108
109 Output_ var = 0;
110 if (skip_nan) {
111 for (Index_ i = 0; i < num_nonzero; ++i, ++value) {
112 auto val = *value;
113 if (!std::isnan(val)) {
114 var += (val - mean) * (val - mean);
115 }
116 }
117 } else {
118 for (Index_ i = 0; i < num_nonzero; ++i, ++value) {
119 auto val = *value;
120 var += (val - mean) * (val - mean);
121 }
122 }
123
124 if (num_nonzero < num_all) {
125 var += (num_all - num_nonzero) * mean * mean;
126 }
127
128 if (count == 0) {
129 return std::make_pair(std::numeric_limits<Output_>::quiet_NaN(), std::numeric_limits<Output_>::quiet_NaN());
130 } else if (count == 1) {
131 return std::make_pair(mean, std::numeric_limits<Output_>::quiet_NaN());
132 } else {
133 return std::make_pair(mean, var / (count - 1));
134 }
135}
