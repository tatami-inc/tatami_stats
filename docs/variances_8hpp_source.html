56template<typename Output_ = double, typename Value_>
57std::pair<Output_, Output_> compute_direct(const Value_* ptr, size_t n) {
58 if (n < 1) {
59 return both_NaN<Output_>();
60 }
61
62 Output_ mean = std::accumulate(ptr, ptr + n, static_cast<Output_>(0))/n;
63 Output_ var = 0;
64 for (size_t j = 0; j < n; ++j, ++ptr) {
65 var += (*ptr - mean) * (*ptr - mean);
66 }
67
68 return std::make_pair(mean, finish_variance_direct(var, n));
69}

85template<typename Output_ = double, typename Value_, typename Index_>
86std::pair<Output_, Output_> compute_direct(const tatami::SparseRange<Value_, Index_>& range, size_t n) {
87 if (n < 1) {
88 return both_NaN<Output_>();
89 }
90
91 Output_ mean = std::accumulate(range.value, range.value + range.number, static_cast<Output_>(0))/n;
92 Output_ var = 0;
93 for (Index_ j = 0; j < range.number; ++j) {
94 var += (range.value[j] - mean) * (range.value[j] - mean);
95 }
96 var += mean * mean * (n - range.number);

150template<typename Value_, typename Index_, typename Output_, typename Nonzero_>
151void compute_running(const tatami::SparseRange<Value_, Index_>& range, Output_* means, Output_* vars, Nonzero_* nonzeros, int& count, bool skip_zeros = true, Index_ subtract = 0) {
152 ++count;
153 for (Index_ j = 0; j < range.number; ++j) {
154 if (!skip_zeros || range.value[j]) {
155 auto ri = range.index[j] - subtract;
156 auto& curM = means[ri];
157 auto& curS = vars[ri];
158 auto& curNZ = nonzeros[ri];
159 ++curNZ;
160
161 const auto& curval = range.value[j];
162 const double delta = curval - curM;
163 curM += delta / curNZ;
164 curS += delta * (curval - curM);
165 }

182template<typename Output_>
183void finish_running(size_t n, Output_* means, Output_* vars, int count) {
184 if (count > 1) {
185 for (size_t i = 0; i < n; ++i) {
186 vars[i] /= count - 1;
187 }
188 } else {
189 if (count == 0){
190 std::fill(means, means + n, std::numeric_limits<double>::quiet_NaN());
191 }

210template<typename Output_, typename Nonzero_>
211void finish_running(size_t n, Output_* means, Output_* vars, const Nonzero_* nonzeros, int count) {
212 if (count) {
213 for (size_t i = 0; i < n; ++i) {
214 const double curNZ = nonzeros[i];
215 const double ratio = curNZ / count;
216 auto& curM = means[i];
217 vars[i] += curM * curM * ratio * (count - curNZ);
218 curM *= ratio;
