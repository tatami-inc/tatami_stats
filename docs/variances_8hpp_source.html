48template<typename Output_ = double, typename Value_, typename Index_ >
49void add_welford(Output_& mean, Output_& sumsq, Value_ value, Index_ count) {
50 Output_ delta = value - mean;
51 mean += delta / count;
52 sumsq += delta * (value - mean);
53}

55template<typename Output_ = double, typename Index_ >
56void add_welford_zeros(Output_& mean, Output_& sumsq, Index_ num_nonzero, Index_ num_all) {
57 auto ratio = static_cast<Output_>(num_nonzero) / static_cast<Output_>(num_all);
58 sumsq += mean * mean * ratio * (num_all - num_nonzero);
59 mean *= ratio;
60}

93template<typename Output_ = double, typename Value_, typename Index_ >
94std::pair<Output_, Output_> direct(const Value_* value, Index_ num_nonzero, Index_ num_all, bool skip_nan) {
95 Output_ mean = 0;
96 Index_ lost = 0;
97
98 ::tatami_stats::internal::nanable_ifelse<Value_>(
99 skip_nan,
100 [&]() {
101 auto copy = value;
102 for (Index_ i = 0; i < num_nonzero; ++i, ++copy) {
103 auto val = *copy;
104 if (std::isnan(val)) {
105 ++lost;
106 } else {
107 mean += val;
108 }
109 }
110 },
111 [&]() {
112 auto copy = value;
113 for (Index_ i = 0; i < num_nonzero; ++i, ++copy) {
114 mean += *copy;
115 }
116 }
117 );
118
119 auto count = num_all - lost;
120 mean /= count;
121
122 Output_ var = 0;
123 ::tatami_stats::internal::nanable_ifelse<Value_>(
124 skip_nan,
125 [&]() {
126 for (Index_ i = 0; i < num_nonzero; ++i, ++value) {
127 auto val = *value;
128 if (!std::isnan(val)) {
129 var += (val - mean) * (val - mean);
130 }
131 }
132 },
133 [&]() {
134 for (Index_ i = 0; i < num_nonzero; ++i, ++value) {
135 auto val = *value;
136 var += (val - mean) * (val - mean);
137 }
138 }
139 );
140
141 if (num_nonzero < num_all) {
142 var += (num_all - num_nonzero) * mean * mean;
143 }
144
145 if (count == 0) {
146 return std::make_pair(std::numeric_limits<Output_>::quiet_NaN(), std::numeric_limits<Output_>::quiet_NaN());
147 } else if (count == 1) {
148 return std::make_pair(mean, std::numeric_limits<Output_>::quiet_NaN());
149 } else {
150 return std::make_pair(mean, var / (count - 1));
151 }
152}

238 if (ct == 0) {

244 }
