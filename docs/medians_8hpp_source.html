32Index_ translocate_nans(Value_* ptr, Index_& num) {
33 Index_ pos = 0;
34 for (Index_ i = 0; i < num; ++i) {
35 if (std::isnan(ptr[i])) {
36 std::swap(ptr[i], ptr[pos]);
37 ++pos;
38 }
39 }
40 return pos;
41}

63template<bool skip_nan_ = false, typename Output_ = double, typename Value_, typename Index_>
64Output_ compute(Value_* ptr, Index_ num) {
65 if constexpr(skip_nan_) {
66 auto lost = internal::translocate_nans(ptr, num);
67 ptr += lost;
68 num -= lost;
69 }
70
71 if (num == 0) {
72 return std::numeric_limits<Output_>::quiet_NaN();
73 }
74
75 size_t halfway = num / 2;
76 bool is_even = (num % 2 == 0);
77
78 // At some point, I found two nth_element calls to be faster than partial_sort.
79 std::nth_element(ptr, ptr + halfway, ptr + num);
80 double medtmp = *(ptr + halfway);
81 if (is_even) {
82 std::nth_element(ptr, ptr + halfway - 1, ptr + num);
83 return (medtmp + *(ptr + halfway - 1))/2;
84 } else {
85 return medtmp;
86 }

106template<bool skip_nan_ = false, typename Output_ = double, typename Value_, typename Index_>
107Output_ compute(Value_* value, Index_ num_nonzero, Index_ num_all) {
108 if constexpr(skip_nan_) {
109 auto lost = internal::translocate_nans(value, num_nonzero);
110 value += lost;
111 num_nonzero -= lost;
112 num_all -= lost;
113 }
114
115 if (num_nonzero == num_all) {
116 return compute<skip_nan_, Output_>(value, num_all);
117 }
118
119 if (num_nonzero * 2 < num_all) {
120 return 0; // zero is the median if there are too many zeroes.
121 }
122
123 size_t halfway = num_all / 2;
124 bool is_even = (num_all % 2 == 0);
125
126 auto vend = value + num_nonzero;
127 std::sort(value, vend);
128 size_t zeropos = std::lower_bound(value, vend, 0) - value;
129 size_t nzero = num_all - num_nonzero;
130
131 if (!is_even) {
132 if (zeropos > halfway) {
133 return value[halfway];
134 } else if (halfway >= zeropos + nzero) {
135 return value[halfway - nzero];
136 } else {
137 return 0; // zero is the median.
138 }
139 }
140
141 double tmp = 0;
142 if (zeropos > halfway) {
143 tmp = value[halfway] + value[halfway - 1];
144 } else if (zeropos == halfway) {
145 // guaranteed to be at least 1 zero.
146 tmp += value[halfway - 1];
147 } else if (zeropos < halfway && zeropos + nzero > halfway) {
148 ; // zero is the median.
149 } else if (zeropos + nzero == halfway) {
150 // guaranteed to be at least 1 zero.
151 tmp += value[halfway - nzero];
152 } else {
153 tmp = value[halfway - nzero] + value[halfway - nzero - 1];
154 }
155 return tmp / 2;
156}

176template<bool skip_nan_, typename Value_, typename Index_, typename Output_>
177void medians(bool row, const tatami::Matrix<Value_, Index_>* p, Output_* output, int threads) {
178 auto dim = (row ? p->nrow() : p->ncol());
179 auto otherdim = (row ? p->ncol() : p->nrow());
180
181 if (p->sparse()) {
182 tatami::Options opt;
183 opt.sparse_extract_index = false;
184 opt.sparse_ordered_index = false; // we'll be sorting by value anyway.
185
186 tatami::parallelize([&](int, Index_ s, Index_ l) -> void {
187 auto ext = tatami::consecutive_extractor<true>(p, row, s, l, opt);
188 std::vector<Value_> buffer(otherdim);
189 auto vbuffer = buffer.data();
190 for (Index_ x = 0; x < l; ++x) {
191 auto range = ext->fetch(vbuffer, NULL);
192 tatami::copy_n(range.value, range.number, vbuffer);
193 output[x + s] = median::compute<skip_nan_, Output_>(vbuffer, range.number, otherdim);
194 }
195 }, dim, threads);
196
197 } else {
198 tatami::parallelize([&](int, Index_ s, Index_ l) -> void {
199 std::vector<Value_> buffer(otherdim);
200 auto ext = tatami::consecutive_extractor<false>(p, row, s, l);
201 for (Index_ x = 0; x < l; ++x) {
202 auto ptr = ext->fetch(buffer.data());
203 tatami::copy_n(ptr, otherdim, buffer.data());
204 output[x + s] = median::compute<skip_nan_, Output_>(buffer.data(), otherdim);
205 }
206 }, dim, threads);
207 }
208}
