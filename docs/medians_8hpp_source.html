60template<typename Output_ = double, typename Value_, typename Index_>
61Output_ direct(Value_* ptr, Index_ num, bool skip_nan) {
62 ::tatami_stats::internal::nanable_ifelse<Value_>(
63 skip_nan,
64 [&]() -> void {
65 auto lost = translocate_nans(ptr, num);
66 ptr += lost;
67 num -= lost;
68 },
69 []() -> void {}
70 );
71
72 if (num == 0) {
73 return std::numeric_limits<Output_>::quiet_NaN();
74 }
75
76 Index_ halfway = num / 2;
77 bool is_even = (num % 2 == 0);
78
79 std::nth_element(ptr, ptr + halfway, ptr + num);
80 Output_ medtmp = *(ptr + halfway);
81 if (!is_even) {
82 return medtmp;
83 }
84
85 // 'nth_element()' reorganizes 'ptr' so that everything below 'halfway' is
86 // less than or equal to 'ptr[halfway]', while everything above 'halfway'
87 // is greater than or equal to 'ptr[halfway]'. Thus, to get the element
88 // immediately before 'halfway' in the sort order, we just need to find the
89 // maximum from '[0, halfway)'.
90 Output_ other = *std::max_element(ptr, ptr + halfway);
91
92 if (medtmp == other) {
93 return medtmp; // Preserve exactness, respect infinities of the same sign.
94 } else {
95 return medtmp + (other - medtmp) / 2; // Avoid FP overflow.
96 }
97}

116template<typename Output_ = double, typename Value_, typename Index_>
117Output_ direct(Value_* value, Index_ num_nonzero, Index_ num_all, bool skip_nan) {
118 // Fallback to the dense code if there are no structural zeros. This is not
119 // just for efficiency as the downstream averaging code assumes that there
120 // is at least one structural zero when considering its scenarios.
121 if (num_nonzero == num_all) {
122 return direct<Output_>(value, num_all, skip_nan);
123 }
124
125 ::tatami_stats::internal::nanable_ifelse<Value_>(
126 skip_nan,
127 [&]() -> void {
128 auto lost = translocate_nans(value, num_nonzero);
129 value += lost;
130 num_nonzero -= lost;
131 num_all -= lost;
132 },
133 []() -> void {}
134 );
135
136 // Is the number of non-zeros less than the number of zeros?
137 // If so, the median must be zero. Note that we calculate it
138 // in this way to avoid overflow from 'num_nonzero * 2'.
139 if (num_nonzero < num_all - num_nonzero) {
140 return 0;
141 }
142
143 Index_ halfway = num_all / 2;
144 bool is_even = (num_all % 2 == 0);
145
146 Index_ num_zero = num_all - num_nonzero;
147 Index_ num_negative = 0;
148 for (Index_ i = 0; i < num_nonzero; ++i) {
149 num_negative += (value[i] < 0);
150 }
151
152 if (!is_even) {
153 if (num_negative > halfway) {
154 std::nth_element(value, value + halfway, value + num_nonzero);
155 return value[halfway];
156
157 } else if (halfway >= num_negative + num_zero) {
158 Index_ skip_zeros = halfway - num_zero;
159 std::nth_element(value, value + skip_zeros, value + num_nonzero);
160 return value[skip_zeros];
161
162 } else {
163 return 0;
164 }
165 }
166
167 Output_ baseline = 0, other = 0;
168 if (num_negative > halfway) { // both halves of the median are negative.
169 std::nth_element(value, value + halfway, value + num_nonzero);
170 baseline = value[halfway];
171 other = *(std::max_element(value, value + halfway)); // max_element gets the sorted value at halfway - 1, see explanation for the dense case.
172
173 } else if (num_negative == halfway) { // the upper half is guaranteed to be zero.
174 Index_ below_halfway = halfway - 1;
175 std::nth_element(value, value + below_halfway, value + num_nonzero);
176 other = value[below_halfway]; // set to other so that addition/subtraction of a zero baseline has no effect on precision.
177
178 } else if (num_negative < halfway && num_negative + num_zero > halfway) { // both halves are zero, so zero is the median.
179 ;
180
181 } else if (num_negative + num_zero == halfway) { // the lower half is guaranteed to be zero.
182 Index_ skip_zeros = halfway - num_zero;
183 std::nth_element(value, value + skip_zeros, value + num_nonzero);
184 other = value[skip_zeros]; // set to other so that addition/subtraction of a zero baseline has no effect on precision.
185
186 } else { // both halves of the median are non-negative.
187 Index_ skip_zeros = halfway - num_zero;
188 std::nth_element(value, value + skip_zeros, value + num_nonzero);
189 baseline = value[skip_zeros];
190 other = *(std::max_element(value, value + skip_zeros)); // max_element gets the sorted value at skip_zeros - 1, see explanation for the dense case.
191 }
192
193 if (baseline == other) {
194 return baseline; // Preserve exactness, respect infinities of the same sign.
195 } else {
196 return baseline + (other - baseline) / 2; // Avoid FP overflow.
197 }
198}

215template<typename Value_, typename Index_, typename Output_>
216void apply(bool row, const tatami::Matrix<Value_, Index_>& mat, Output_* output, const medians::Options& mopt) {
217 auto dim = (row ? mat.nrow() : mat.ncol());
218 auto otherdim = (row ? mat.ncol() : mat.nrow());
219
220 if (mat.sparse()) {
221 tatami::Options opt;
222 opt.sparse_extract_index = false;
223 opt.sparse_ordered_index = false; // we'll be sorting by value anyway.
224
225 tatami::parallelize([&](int, Index_ s, Index_ l) -> void {
226 auto ext = tatami::consecutive_extractor<true>(mat, row, s, l, opt);
227 auto buffer = tatami::create_container_of_Index_size<std::vector<Value_> >(otherdim);
228 auto vbuffer = buffer.data();
229 for (Index_ x = 0; x < l; ++x) {
230 auto range = ext->fetch(vbuffer, NULL);
231 tatami::copy_n(range.value, range.number, vbuffer);
232 output[x + s] = medians::direct<Output_>(vbuffer, range.number, otherdim, mopt.skip_nan);
233 }
234 }, dim, mopt.num_threads);
235
236 } else {
237 tatami::parallelize([&](int, Index_ s, Index_ l) -> void {
238 auto buffer = tatami::create_container_of_Index_size<std::vector<Value_> >(otherdim);
239 auto ext = tatami::consecutive_extractor<false>(mat, row, s, l);
240 for (Index_ x = 0; x < l; ++x) {
241 auto ptr = ext->fetch(buffer.data());
242 tatami::copy_n(ptr, otherdim, buffer.data());
243 output[x + s] = medians::direct<Output_>(buffer.data(), otherdim, mopt.skip_nan);
244 }
245 }, dim, mopt.num_threads);
246 }
247}
