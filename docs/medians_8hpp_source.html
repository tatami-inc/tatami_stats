62template<typename Output_ = double, typename Value_, typename Index_>
63Output_ direct(Value_* value, Index_ num_nonzero, Index_ num_all, bool skip_nan) {
64 ::tatami_stats::internal::nanable_ifelse<Value_>(
65 skip_nan,
66 [&]() -> void {
67 auto new_nonzero = shift_nans(value, num_nonzero);
68 num_all -= num_nonzero - new_nonzero;
69 num_nonzero = new_nonzero;
70 },
71 []() -> void {}
72 );
73
74 return quickstats::median<Output_>(num_all, num_nonzero, value);
75}
95template<typename Value_, typename Index_, typename Output_>
96void apply(bool row, const tatami::Matrix<Value_, Index_>& mat, Output_* output, const medians::Options& mopt) {
97 auto dim = (row ? mat.nrow() : mat.ncol());
98 auto otherdim = (row ? mat.ncol() : mat.nrow());
99
100 if (mat.sparse()) {
101 tatami::Options opt;
102 opt.sparse_extract_index = false;
103 opt.sparse_ordered_index = false; // we'll be sorting by value anyway.
104
105 tatami::parallelize([&](int, Index_ s, Index_ l) -> void {
106 auto ext = tatami::consecutive_extractor<true>(mat, row, s, l, opt);
107 auto buffer = tatami::create_container_of_Index_size<std::vector<Value_> >(otherdim);
108 auto vbuffer = buffer.data();
109 for (Index_ x = 0; x < l; ++x) {
110 auto range = ext->fetch(vbuffer, NULL);
111 tatami::copy_n(range.value, range.number, vbuffer);
112 output[x + s] = direct<Output_>(vbuffer, range.number, otherdim, mopt.skip_nan);
113 }
114 }, dim, mopt.num_threads);
115
116 } else {
117 tatami::parallelize([&](int, Index_ s, Index_ l) -> void {
118 auto buffer = tatami::create_container_of_Index_size<std::vector<Value_> >(otherdim);
119 auto ext = tatami::consecutive_extractor<false>(mat, row, s, l);
120 for (Index_ x = 0; x < l; ++x) {
121 auto ptr = ext->fetch(buffer.data());
122 tatami::copy_n(ptr, otherdim, buffer.data());
123 output[x + s] = direct<Output_>(buffer.data(), otherdim, mopt.skip_nan);
124 }
125 }, dim, mopt.num_threads);
126 }
127}
