47template<typename Value_, typename Index_>
48Index_ translocate_nans(Value_* ptr, Index_& num) {
49 Index_ pos = 0;
50 for (Index_ i = 0; i < num; ++i) {
51 if (std::isnan(ptr[i])) {
52 std::swap(ptr[i], ptr[pos]);
53 ++pos;
54 }
55 }
56 return pos;
57}

79template<typename Output_ = double, typename Value_, typename Index_>
80Output_ direct(Value_* ptr, Index_ num, bool skip_nan) {
81 if (skip_nan) {
82 auto lost = internal::translocate_nans(ptr, num);
83 ptr += lost;
84 num -= lost;
85 }
86
87 if (num == 0) {
88 return std::numeric_limits<Output_>::quiet_NaN();
89 }
90
91 size_t halfway = num / 2;
92 bool is_even = (num % 2 == 0);
93
94 // At some point, I found two nth_element calls to be faster than partial_sort.
95 std::nth_element(ptr, ptr + halfway, ptr + num);
96 double medtmp = *(ptr + halfway);
97 if (is_even) {
98 std::nth_element(ptr, ptr + halfway - 1, ptr + num);
99 return (medtmp + *(ptr + halfway - 1))/2;
100 } else {
101 return medtmp;
102 }
103}

122template<typename Output_ = double, typename Value_, typename Index_>
123Output_ direct(Value_* value, Index_ num_nonzero, Index_ num_all, bool skip_nan) {
124 if (num_nonzero == num_all) {
125 return direct<Output_>(value, num_all, skip_nan);
126 }
127
128 if (skip_nan) {
129 auto lost = internal::translocate_nans(value, num_nonzero);
130 value += lost;
131 num_nonzero -= lost;
132 num_all -= lost;
133 }
134
135 if (num_nonzero * 2 < num_all) {
136 return 0; // zero is the median if there are too many zeroes.
137 }
138
139 size_t halfway = num_all / 2;
140 bool is_even = (num_all % 2 == 0);
141
142 auto vend = value + num_nonzero;
143 std::sort(value, vend);
144 size_t zeropos = std::lower_bound(value, vend, 0) - value;
145 size_t nzero = num_all - num_nonzero;
146
147 if (!is_even) {
148 if (zeropos > halfway) {
149 return value[halfway];
150 } else if (halfway >= zeropos + nzero) {
151 return value[halfway - nzero];
152 } else {
153 return 0; // zero is the median.
154 }
155 }
156
157 double tmp = 0;
158 if (zeropos > halfway) {
159 tmp = value[halfway] + value[halfway - 1];
160 } else if (zeropos == halfway) {
161 // guaranteed to be at least 1 zero.
162 tmp += value[halfway - 1];
163 } else if (zeropos < halfway && zeropos + nzero > halfway) {
164 ; // zero is the median.
165 } else if (zeropos + nzero == halfway) {
166 // guaranteed to be at least 1 zero.
167 tmp += value[halfway - nzero];
168 } else {
169 tmp = value[halfway - nzero] + value[halfway - nzero - 1];
170 }
171 return tmp / 2;
172}

188template<typename Value_, typename Index_, typename Output_>
189void apply(bool row, const tatami::Matrix<Value_, Index_>* p, Output_* output, const medians::Options& mopt) {
190 auto dim = (row ? p->nrow() : p->ncol());
191 auto otherdim = (row ? p->ncol() : p->nrow());
192
193 if (p->sparse()) {
194 tatami::Options opt;
195 opt.sparse_extract_index = false;
196 opt.sparse_ordered_index = false; // we'll be sorting by value anyway.
197
198 tatami::parallelize([&](int, Index_ s, Index_ l) -> void {
199 auto ext = tatami::consecutive_extractor<true>(p, row, s, l, opt);
200 std::vector<Value_> buffer(otherdim);
201 auto vbuffer = buffer.data();
202 for (Index_ x = 0; x < l; ++x) {
203 auto range = ext->fetch(vbuffer, NULL);
204 tatami::copy_n(range.value, range.number, vbuffer);
205 output[x + s] = medians::direct<Output_>(vbuffer, range.number, otherdim, mopt.skip_nan);
206 }
207 }, dim, mopt.num_threads);
208
209 } else {
210 tatami::parallelize([&](int, Index_ s, Index_ l) -> void {
211 std::vector<Value_> buffer(otherdim);
212 auto ext = tatami::consecutive_extractor<false>(p, row, s, l);
213 for (Index_ x = 0; x < l; ++x) {
214 auto ptr = ext->fetch(buffer.data());
215 tatami::copy_n(ptr, otherdim, buffer.data());
216 output[x + s] = medians::direct<Output_>(buffer.data(), otherdim, mopt.skip_nan);
217 }
218 }, dim, mopt.num_threads);
219 }
220}
