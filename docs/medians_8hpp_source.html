32Index_ translocate_nans(Value_* ptr, Index_& num) {
33 Index_ pos = 0;
34 for (Index_ i = 0; i < num; ++i) {
35 if (std::isnan(ptr[i])) {
36 std::swap(ptr[i], ptr[pos]);
37 ++pos;
38 }
39 }
40 return pos;
41}

62template<bool skip_nan_ = false, typename Output_ = double, typename Value_, typename Index_>
63Output_ compute(Value_* ptr, Index_ num) {
64 if constexpr(skip_nan_) {
65 auto lost = internal::translocate_nans(ptr, num);
66 ptr += lost;
67 num -= lost;
68 }
69
70 if (num == 0) {
71 return std::numeric_limits<Output_>::quiet_NaN();
72 }
73
74 size_t halfway = num / 2;
75 bool is_even = (num % 2 == 0);
76
77 // At some point, I found two nth_element calls to be faster than partial_sort.
78 std::nth_element(ptr, ptr + halfway, ptr + num);
79 double medtmp = *(ptr + halfway);
80 if (is_even) {
81 std::nth_element(ptr, ptr + halfway - 1, ptr + num);
82 return (medtmp + *(ptr + halfway - 1))/2;
83 } else {
84 return medtmp;
85 }

104template<bool skip_nan_ = false, typename Output_ = double, typename Value_, typename Index_>
105Output_ compute(Value_* value, Index_ num_nonzero, Index_ num_all) {
106 if constexpr(skip_nan_) {
107 auto lost = internal::translocate_nans(value, num_nonzero);
108 value += lost;
109 num_nonzero -= lost;
110 num_all -= lost;
111 }
112
113 if (num_nonzero == num_all) {
114 return compute<skip_nan_, Output_>(value, num_all);
115 }
116
117 if (num_nonzero * 2 < num_all) {
118 return 0; // zero is the median if there are too many zeroes.
119 }
120
121 size_t halfway = num_all / 2;
122 bool is_even = (num_all % 2 == 0);
123
124 auto vend = value + num_nonzero;
125 std::sort(value, vend);
126 size_t zeropos = std::lower_bound(value, vend, 0) - value;
127 size_t nzero = num_all - num_nonzero;
128
129 if (!is_even) {
130 if (zeropos > halfway) {
131 return value[halfway];
132 } else if (halfway >= zeropos + nzero) {
133 return value[halfway - nzero];
134 } else {
135 return 0; // zero is the median.
136 }
137 }
138
139 double tmp = 0;
140 if (zeropos > halfway) {
141 tmp = value[halfway] + value[halfway - 1];
142 } else if (zeropos == halfway) {
143 // guaranteed to be at least 1 zero.
144 tmp += value[halfway - 1];
145 } else if (zeropos < halfway && zeropos + nzero > halfway) {
146 ; // zero is the median.
147 } else if (zeropos + nzero == halfway) {
148 // guaranteed to be at least 1 zero.
149 tmp += value[halfway - nzero];
150 } else {
151 tmp = value[halfway - nzero] + value[halfway - nzero - 1];
152 }
153 return tmp / 2;
154}

172template<bool skip_nan_, typename Value_, typename Index_, typename Output_>
173void medians(bool row, const tatami::Matrix<Value_, Index_>* p, Output_* output, int threads) {
174 auto dim = (row ? p->nrow() : p->ncol());
175 auto otherdim = (row ? p->ncol() : p->nrow());
176
177 if (p->sparse()) {
178 tatami::Options opt;
179 opt.sparse_extract_index = false;
180 opt.sparse_ordered_index = false; // we'll be sorting by value anyway.
181
182 tatami::parallelize([&](int, Index_ s, Index_ l) -> void {
183 auto ext = tatami::consecutive_extractor<true>(p, row, s, l, opt);
184 std::vector<Value_> buffer(otherdim);
185 auto vbuffer = buffer.data();
186 for (Index_ x = 0; x < l; ++x) {
187 auto range = ext->fetch(vbuffer, NULL);
188 tatami::copy_n(range.value, range.number, vbuffer);
189 output[x + s] = median::compute<skip_nan_, Output_>(vbuffer, range.number, otherdim);
190 }
191 }, dim, threads);
192
193 } else {
194 tatami::parallelize([&](int, Index_ s, Index_ l) -> void {
195 std::vector<Value_> buffer(otherdim);
196 auto ext = tatami::consecutive_extractor<false>(p, row, s, l);
197 for (Index_ x = 0; x < l; ++x) {
198 auto ptr = ext->fetch(buffer.data());
199 tatami::copy_n(ptr, otherdim, buffer.data());
200 output[x + s] = median::compute<skip_nan_, Output_>(buffer.data(), otherdim);
201 }
202 }, dim, threads);
203 }
204}
