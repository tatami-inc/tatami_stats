49template<typename Value_, typename Index_>
50Index_ translocate_nans(Value_* ptr, Index_& num) {
51 Index_ pos = 0;
52 for (Index_ i = 0; i < num; ++i) {
53 if (std::isnan(ptr[i])) {
54 std::swap(ptr[i], ptr[pos]);
55 ++pos;
56 }
57 }
58 return pos;
59}

81template<typename Output_ = double, typename Value_, typename Index_>
82Output_ direct(Value_* ptr, Index_ num, bool skip_nan) {
83 ::tatami_stats::internal::nanable_ifelse<Value_>(
84 skip_nan,
85 [&]() {
86 auto lost = internal::translocate_nans(ptr, num);
87 ptr += lost;
88 num -= lost;
89 },
90 [&]() {}
91 );
92
93 if (num == 0) {
94 return std::numeric_limits<Output_>::quiet_NaN();
95 }
96
97 size_t halfway = num / 2;
98 bool is_even = (num % 2 == 0);
99
100 std::nth_element(ptr, ptr + halfway, ptr + num);
101 Output_ medtmp = *(ptr + halfway);
102 if (!is_even) {
103 return medtmp;
104 }
105
106 // 'nth_element()' reorganizes 'ptr' so that everything below 'halfway' is
107 // less than or equal to 'ptr[halfway]', while everything above 'halfway'
108 // is greater than or equal to 'ptr[halfway]'. Thus, to get the element
109 // immediately before 'halfway' in the sort order, we just need to find the
110 // maximum from '[0, halfway)'.
111 Output_ other = *std::max_element(ptr, ptr + halfway);
112
113 // Respect infinities of the same sign.
114 if (std::isinf(medtmp) && medtmp == other) {
115 return medtmp;
116 }
117
118 // Avoid FP overflow, preserve exactness of the median if medtmp == other.
119 return medtmp + (other - medtmp)/2;
120}

139template<typename Output_ = double, typename Value_, typename Index_>
140Output_ direct(Value_* value, Index_ num_nonzero, Index_ num_all, bool skip_nan) {
141 // Fallback to the dense code if there are no structural zeros. This is not
142 // just for efficiency as the downstream averaging code assumes that there
143 // is at least one structural zero when considering its scenarios.
144 if (num_nonzero == num_all) {
145 return direct<Output_>(value, num_all, skip_nan);
146 }
147
148 ::tatami_stats::internal::nanable_ifelse<Value_>(
149 skip_nan,
150 [&]() {
151 auto lost = internal::translocate_nans(value, num_nonzero);
152 value += lost;
153 num_nonzero -= lost;
154 num_all -= lost;
155 },
156 [&]() {}
157 );
158
159 // Is the number of non-zeros less than the number of zeros?
160 // If so, the median must be zero. Note that we calculate it
161 // in this way to avoid overflow from 'num_nonzero * 2'.
162 if (num_nonzero < num_all - num_nonzero) {
163 return 0;
164 }
165
166 size_t halfway = num_all / 2;
167 bool is_even = (num_all % 2 == 0);
168
169 size_t num_zero = num_all - num_nonzero;
170 size_t num_negative = 0;
171 for (Index_ i = 0; i < num_nonzero; ++i) {
172 num_negative += (value[i] < 0);
173 }
174
175 if (!is_even) {
176 if (num_negative > halfway) {
177 std::nth_element(value, value + halfway, value + num_nonzero);
178 return value[halfway];
179
180 } else if (halfway >= num_negative + num_zero) {
181 size_t skip_zeros = halfway - num_zero;
182 std::nth_element(value, value + skip_zeros, value + num_nonzero);
183 return value[skip_zeros];
184
185 } else {
186 return 0;
187 }
188 }
189
190 Output_ baseline = 0, other = 0;
191 if (num_negative > halfway) { // both halves of the median are negative.
192 std::nth_element(value, value + halfway, value + num_nonzero);
193 baseline = value[halfway];
194 other = *(std::max_element(value, value + halfway)); // max_element gets the sorted value at halfway - 1, see explanation for the dense case.
195
196 } else if (num_negative == halfway) { // the upper half is guaranteed to be zero.
197 size_t below_halfway = halfway - 1;
198 std::nth_element(value, value + below_halfway, value + num_nonzero);
199 other = value[below_halfway]; // set to other so that addition/subtraction of a zero baseline has no effect on precision.
200
201 } else if (num_negative < halfway && num_negative + num_zero > halfway) { // both halves are zero, so zero is the median.
202 ;
203
204 } else if (num_negative + num_zero == halfway) { // the lower half is guaranteed to be zero.
205 size_t skip_zeros = halfway - num_zero;
206 std::nth_element(value, value + skip_zeros, value + num_nonzero);
207 other = value[skip_zeros]; // set to other so that addition/subtraction of a zero baseline has no effect on precision.
208
209 } else { // both halves of the median are non-negative.
210 size_t skip_zeros = halfway - num_zero;
211 std::nth_element(value, value + skip_zeros, value + num_nonzero);
212 baseline = value[skip_zeros];
213 other = *(std::max_element(value, value + skip_zeros)); // max_element gets the sorted value at skip_zeros - 1, see explanation for the dense case.
214 }
215
216 // Respect infinities of the same sign.
217 if (std::isinf(baseline) && baseline == other) {
218 return baseline;
219 }
220
221 // Avoid FP overflow, preserve exactness of the median if baseline == other.
222 return baseline + (other - baseline) / 2;
223}

240template<typename Value_, typename Index_, typename Output_>
241void apply(bool row, const tatami::Matrix<Value_, Index_>* p, Output_* output, const medians::Options& mopt) {
242 auto dim = (row ? p->nrow() : p->ncol());
243 auto otherdim = (row ? p->ncol() : p->nrow());
244
245 if (p->sparse()) {
246 tatami::Options opt;
247 opt.sparse_extract_index = false;
248 opt.sparse_ordered_index = false; // we'll be sorting by value anyway.
249
250 tatami::parallelize([&](int, Index_ s, Index_ l) -> void {
251 auto ext = tatami::consecutive_extractor<true>(p, row, s, l, opt);
252 std::vector<Value_> buffer(otherdim);
253 auto vbuffer = buffer.data();
254 for (Index_ x = 0; x < l; ++x) {
255 auto range = ext->fetch(vbuffer, NULL);
256 tatami::copy_n(range.value, range.number, vbuffer);
257 output[x + s] = medians::direct<Output_>(vbuffer, range.number, otherdim, mopt.skip_nan);
258 }
259 }, dim, mopt.num_threads);
260
261 } else {
262 tatami::parallelize([&](int, Index_ s, Index_ l) -> void {
263 std::vector<Value_> buffer(otherdim);
264 auto ext = tatami::consecutive_extractor<false>(p, row, s, l);
265 for (Index_ x = 0; x < l; ++x) {
266 auto ptr = ext->fetch(buffer.data());
267 tatami::copy_n(ptr, otherdim, buffer.data());
268 output[x + s] = medians::direct<Output_>(buffer.data(), otherdim, mopt.skip_nan);
269 }
270 }, dim, mopt.num_threads);
271 }
272}
